class Solution {

public:

    ListNode* reverseList(ListNode* head) {

        ListNode*prev = nullptr;

        ListNode*cur = head;

        while(cur!=nullptr){

            ListNode*nexted = cur->next;

            cur->next = prev;

            prev = cur;

            cur = nexted;

        }

        return prev;

    }

};
